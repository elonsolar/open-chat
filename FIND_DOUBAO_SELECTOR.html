🔍 豆包选择器检测工具

帮助找到豆包的正确选择器

📋 操作步骤

  1. 打开豆包网站并登录
  2. 按F12打开开发者工具
  3. 切换到Console标签
  4. 依次运行下面的检测代码
  5. 把结果截图发给我

1️⃣ 检测1：输入框

// 检测输入框 
const textarea = document.querySelector('textarea'); console.log('找到输入框:', !!textarea); if (textarea) { console.log('输入框类名:', textarea.className); console.log('输入框id:', textarea.id); console.log('输入框placeholder:', textarea.placeholder); } else { console.log('未找到textarea，尝试其他方式'); }

2️⃣ 检测2：查找所有消息容器

// 查找消息容器 
// 尝试不同的消息容器选择器 const selectors = [ '[class*="message"]', '[class*="chat"]', '[class*="conversation"]', '[class*="dialog"]', '.message', '.chat', '.conversation', '.dialog' ]; selectors.forEach(sel => { const elements = document.querySelectorAll(sel); console.log(`选择器 "${sel}":`, elements.length, '个'); });

3️⃣ 检测3：查看第一个消息

// 查找第一个消息的结构 
// 尝试多种方式查找 const all = document.querySelectorAll('*'); console.log('页面总元素:', all.length); // 查找包含特定文本的元素 const textElements = Array.from(all).filter(el => { const text = el.textContent?.trim(); return text && text.length > 10 && text.length < 100; }); console.log('可能的文本元素:', textElements.length); // 如果找到了，显示第一个的结构 if (textElements.length > 0) { const first = textElements[0]; console.log('第一个文本元素:'); console.log(' 标签:', first.tagName); console.log(' 类名:', first.className); console.log(' HTML:', first.outerHTML.substring(0, 200)); }

4️⃣ 完整诊断脚本

// 完整诊断 
console.log('=== 豆包DOM结构诊断 ==='); // 1. 基本信息 console.log('URL:', window.location.href); console.log('标题:', document.title); // 2. 查找输入框 const input = document.querySelector('textarea'); if (input) { console.log('✓ 找到输入框'); console.log(' 类名:', input.className); } else { console.log('✗ 未找到textarea'); } // 3. 查找按钮 const buttons = document.querySelectorAll('button, [role="button"]'); console.log('✓ 按钮/可点击元素数量:', buttons.length); // 4. 查找消息容器 const messageContainers = [ document.querySelectorAll('[class*="message"]').length, document.querySelectorAll('[class*="chat"]').length, document.querySelectorAll('[class*="conversation"]').length, document.querySelectorAll('.message').length ]; console.log('消息容器数量:', messageContainers); // 5. 查找所有可能的用户消息 const userMessages = document.querySelectorAll('[class*="user"], [class*="human"]'); console.log('用户消息元素:', userMessages.length); // 6. 查找所有可能的AI消息 const aiMessages = document.querySelectorAll('[class*="assistant"], [class*="ai"], [class*="bot"]'); console.log('AI消息元素:', aiMessages.length); // 7. 查找所有文本元素 const textElements = document.querySelectorAll('p, div, span, section, article'); const textWithContent = Array.from(textElements).filter(el => { const text = el.textContent?.trim(); return text && text.length > 10 && text.length < 1000; }); console.log('有文本的元素数量:', textWithContent.length); console.log('=== 诊断完成 ===');
